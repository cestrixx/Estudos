Contas_a_pagar - Cadastrar uma despesa

2.0.0 
Requisição para cadastro de despesas

POST
https://api.vhsys.com/v2/contas-pagar
cURL
PHP
curl -X POST \
    https://{api_address}/v2/contas-pagar \
    -H 'access-token: SEU (Access-Token) DE ACESSO ' \
    -H 'secret-access-token: SEU (Secret-Access-Token) DE ACESSO' \
    -H 'cache-control: no-cache' \
    -H 'content-type: application/json' \
    -d '{
        "nome_conta" : "Nome da conta",
        "id_categoria" : "0000",
        "categoria_pag" : "0000",
        "id_banco" : "000",
        "id_fornecedor" : "00",
        "nome_fornecedor" : "Nome do fornecedor",
        "vencimento_pag" : "0000-00-00",
        "valor_pag" : "00.00",
        "valor_pago" : "00.00",
        "data_emissao" : "0000-00-00",
        "n_documento_pag" : "0000",
        "observacoes_pag" : "Observações",
        "id_centro_custos" : "000",
        "centro_custos_pag" : "Centro de custo",
        "liquidado_pag" : "Sim",
        "data_pagamento" : "0000-00-00",

    -H 'access-token: SEU (Access-Token) DE ACESSO ' \
    -H 'secret-access-token: SEU (Secret-Access-Token) DE ACESSO' \
    -H 'cache-control: no-cache' \
